        public int AccountBalanace(int accountbalance, int depositeamount)

        {
            if (balance >= 0)

                balance -= moneytowithdraw;
                Console.WriteLine($"The withdrawl amount ----> {moneytowithdraw}");
                Console.WriteLine($"The Account Balance after withdrawl is ----> {balance}");
            }
            else
            {
                Console.WriteLine("No suffient money to withdraw");
            }
            return balance;
        }
